void main()
{
	int a[20],i,j,n,temp,pos,ch;
	clrscr();
	cout<<"Enter the size of Array:"<<endl;
	cin>>n;
	cout<<"Enter Array Elements:\n";
	for(i=0;i<n;i++)
	{
		cin>>a[i];

	}
	/*Sorting the array*/
	for(i=0;i<=n-1;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			if(a[j+1]>a[j])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	cout<<"Sorted Array:\n";
	for(i=0;i<n;i++)
	{
		cout<<a[i]<<"  ";
	}
	while(1)

		cin>>ch;

				cin>>pos;
				cout<<a[pos];
				break;
			case 2: return;
			default:cout<<"Wrong choice\n";
				break;
		}
	getch();
	}
}
